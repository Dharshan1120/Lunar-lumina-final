  <script>
    // ── Three.js Scene ──
    let scene, camera, renderer, currentMode = 'particles';
    let particles, waveMesh, nebulaGroup;
    let mouseX = 0, mouseY = 0;
    let frameCount = 0, lastFpsTime = Date.now();

    function initThree() {
      const canvas = document.getElementById('three-canvas');
      renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: true });
      renderer.setSize(window.innerWidth, window.innerHeight);
      renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

      scene = new THREE.Scene();
      camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 1000);
      camera.position.z = 5;

      createParticles();
      createWaves();
      createNebula();

      showMode('particles');

      window.addEventListener('resize', onResize);
      document.addEventListener('mousemove', (e) => {
        mouseX = (e.clientX / window.innerWidth - 0.5) * 2;
        mouseY = (e.clientY / window.innerHeight - 0.5) * 2;
      });

      animate();
    }

    function createParticles() {
      const count = 5000;
      const geo = new THREE.BufferGeometry();
      const pos = new Float32Array(count * 3);
      const colors = new Float32Array(count * 3);
      const sizes = new Float32Array(count);

      for (let i = 0; i < count; i++) {
        const i3 = i * 3;
        const theta = Math.random() * Math.PI * 2;
        const phi = Math.acos(2 * Math.random() - 1);
        const r = 2.5 + Math.random() * 2;
        pos[i3] = r * Math.sin(phi) * Math.cos(theta);
        pos[i3 + 1] = r * Math.sin(phi) * Math.sin(theta);
        pos[i3 + 2] = r * Math.cos(phi);

        // Indigo-purple palette
        const t = Math.random();
        colors[i3] = 0.4 + t * 0.35;      // R
        colors[i3 + 1] = 0.4 + t * 0.25;  // G
        colors[i3 + 2] = 0.95 + t * 0.05; // B
        sizes[i] = Math.random() * 3 + 1;
      }

      geo.setAttribute('position', new THREE.BufferAttribute(pos, 3));
      geo.setAttribute('color', new THREE.BufferAttribute(colors, 3));
      geo.setAttribute('size', new THREE.BufferAttribute(sizes, 1));

      const mat = new THREE.PointsMaterial({
        size: 0.035,
        vertexColors: true,
        transparent: true,
        opacity: 0.8,
        blending: THREE.AdditiveBlending,
        depthWrite: false,
      });

      particles = new THREE.Points(geo, mat);
      scene.add(particles);
    }

    function createWaves() {
      const geo = new THREE.PlaneGeometry(12, 12, 128, 128);
      const mat = new THREE.MeshBasicMaterial({
        color: 0x6366f1,
        wireframe: true,
        transparent: true,
        opacity: 0.3,
      });
      waveMesh = new THREE.Mesh(geo, mat);
      waveMesh.rotation.x = -Math.PI / 2.5;
      waveMesh.position.y = -1;
      waveMesh.visible = false;
      scene.add(waveMesh);
    }

    function createNebula() {
      nebulaGroup = new THREE.Group();
      const count = 80;
      for (let i = 0; i < count; i++) {
        const geo = new THREE.SphereGeometry(0.15 + Math.random() * 0.6, 8, 8);
        const mat = new THREE.MeshBasicMaterial({
          color: new THREE.Color().setHSL(0.68 + Math.random() * 0.12, 0.8, 0.45 + Math.random() * 0.2),
          transparent: true,
          opacity: 0.08 + Math.random() * 0.12,
          blending: THREE.AdditiveBlending,
          depthWrite: false,
        });
        const mesh = new THREE.Mesh(geo, mat);
        mesh.position.set(
          (Math.random() - 0.5) * 8,
          (Math.random() - 0.5) * 6,
          (Math.random() - 0.5) * 4
        );
        mesh.userData.speed = 0.2 + Math.random() * 0.5;
        mesh.userData.offset = Math.random() * Math.PI * 2;
        nebulaGroup.add(mesh);
      }
      nebulaGroup.visible = false;
      scene.add(nebulaGroup);
    }

    function showMode(mode) {
      currentMode = mode;
      particles.visible = mode === 'particles';
      waveMesh.visible = mode === 'waves';
      nebulaGroup.visible = mode === 'nebula';

      const labels = { particles: '5K', waves: '16K', nebula: '80' };
      document.getElementById('particleCount').textContent = labels[mode];
    }

    function animate() {
      requestAnimationFrame(animate);
      const t = performance.now() * 0.001;

      // Smooth camera follow
      camera.position.x += (mouseX * 0.8 - camera.position.x) * 0.03;
      camera.position.y += (-mouseY * 0.5 - camera.position.y) * 0.03;
      camera.lookAt(0, 0, 0);

      if (particles.visible) {
        particles.rotation.y = t * 0.08;
        particles.rotation.x = Math.sin(t * 0.05) * 0.15;
        const pos = particles.geometry.attributes.position.array;
        for (let i = 0; i < pos.length; i += 3) {
          pos[i + 1] += Math.sin(t * 0.8 + pos[i] * 0.5) * 0.001;
        }
        particles.geometry.attributes.position.needsUpdate = true;
      }

      if (waveMesh.visible) {
        const pos = waveMesh.geometry.attributes.position.array;
        for (let i = 0; i < pos.length; i += 3) {
          const x = pos[i], z = pos[i + 2] || 0;
          pos[i + 2] = Math.sin(x * 0.6 + t * 1.2) * Math.cos(z * 0.4 + t * 0.8) * 0.8;
        }
        waveMesh.geometry.attributes.position.needsUpdate = true;
        waveMesh.rotation.z = t * 0.02;
      }

      if (nebulaGroup.visible) {
        nebulaGroup.children.forEach((m) => {
          m.position.x += Math.sin(t * m.userData.speed + m.userData.offset) * 0.003;
          m.position.y += Math.cos(t * m.userData.speed * 0.7 + m.userData.offset) * 0.002;
          const s = 1 + Math.sin(t * m.userData.speed + m.userData.offset) * 0.15;
          m.scale.set(s, s, s);
        });
        nebulaGroup.rotation.y = t * 0.03;
      }

      renderer.render(scene, camera);

      // FPS counter
      frameCount++;
      if (Date.now() - lastFpsTime >= 1000) {
        document.getElementById('fpsCounter').textContent = frameCount;
        frameCount = 0;
        lastFpsTime = Date.now();
      }
    }

    function onResize() {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
    }

    // ── Mode Switching ──
    document.querySelectorAll('.mode-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        document.querySelectorAll('.mode-btn').forEach(b => {
          b.classList.remove('active');
        });
        btn.classList.add('active');
        showMode(btn.dataset.mode);
      });
    });

    // ── Init ──
    initThree();
    lucide.createIcons();
  </script>
